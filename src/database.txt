create table kullanicilar( can
	id serial primary key,
	kullanici_adi varchar(20),
	sifre varchar(16) not null,
	sehir varchar(30),
	yas int,
	cinsiyet bool,
	sef bool default false
);

create table yetkililer( furkan
	id int primary key references kullanicilar(id)
);

create table yorum_puan( can
	id serial primary key,
	kullanici int references kullananicilar(id)
	yorum varchar(200),
	puan int
);

create table ana_yemekler(  furkan
	id serial primary key,
	yemek_adi varchar(30),
	tarif text,
	malzemeler text,
	kac_kisilik int,
	hazirlama_sure int,
	pisirme_sure int,
	sef int references kullanicilar(id),
	yorum_puan int references yorum_puan(id)
);
